/* ── PARTICLES ── */
(function(){
  const c = document.getElementById('particles');
  for(let i = 0; i < 55; i++){
    const p = document.createElement('div');
    p.className = 'particle';
    const sz = Math.random() * 3 + 1;

    p.style.cssText = `

      background:rgba(${hue},.7);
      left:${Math.random() * 100}%;
      animation-duration:${8 + Math.random() * 14}s;

/* ── COUNTDOWN + BUTTON DELAY ── */
const DELAY = 90;           // seconds — change to match your video unlock point
let remaining = DELAY;
let started   = false;
let interval  = null;

const btn   = document.getElementById('ctaBtn');
const cd    = document.getElementById('countdown');
const timer = document.getElementById('timer');
const label = document.getElementById('ctaLabel');
const ring  = document.getElementById('timerRing');
const ringW = document.getElementById('ringWrap');

function fmt(s){
  const m = Math.floor(s / 60), sec = s % 60;
  return `${m}:${sec < 10 ? '0' : ''}${sec}`;
}

// Perimeter of the rect (296 × 56)
const perim = 2 * (296 + 56);
ring.style.strokeDasharray = `0 ${perim}`;

function startCountdown(){
  if(started) return;
  started = true;
  ringW.style.display = 'block';
  label.classList.add('active');

  interval = setInterval(() => {
    remaining--;
    timer.textContent = fmt(remaining);

    // Ring fills up as time passes
    const progress = ((DELAY - remaining) / DELAY) * perim;
    ring.style.strokeDasharray = `${progress} ${perim}`;

    if(remaining <= 0){
      clearInterval(interval);
      unlock();
    }
  }, 1000);
}

function unlock(){
  btn.disabled = false;
  cd.classList.add('hidden');
  label.textContent = 'You are one step away from changing everything.';
  label.style.color         = 'var(--gold)';
  label.style.opacity       = '1';
  label.style.letterSpacing = '.1em';

  // Pulse entrance animation
  btn.animate([
    { boxShadow: '0 0 0 0 rgba(32,96,200,.6)' },
    { boxShadow: '0 0 0 18px rgba(32,96,200,0)' }
  ], { duration: 900, iterations: 3 });
}

/* ── Auto-start countdown when the video section enters the viewport ── */
const obs = new IntersectionObserver(entries => {
  if(entries[0].isIntersecting){ startCountdown(); obs.disconnect(); }
}, { threshold: .4 });
obs.observe(document.getElementById('vsec'));

/* ── Placeholder click (simulates video play) ── */
window.startVideo = function(){
  const ph = document.querySelector('.vid-placeholder');
  if(ph) ph.style.display = 'none';
  startCountdown();
  // If using YouTube IFrame API, replace this with onPlayerStateChange = PLAYING
};

/* ── CTA button redirect ── */

  window.location.href = 'https://sofiadeluca.com/'; // ← replace with your destination URL
});

## Customization Cheatsheet

### 1. Add your video

**YouTube:**
```html
<!-- Remove the .vid-placeholder div and uncomment this inside .vid-wrap -->
<iframe id="yt"
  src="https://www.youtube.com/embed/YOUR_VIDEO_ID?enablejsapi=1&rel=0&modestbranding=1"
  allow="autoplay; encrypted-media" allowfullscreen></iframe>
```

**Vimeo:**
```html
<iframe src="https://player.vimeo.com/video/YOUR_VIDEO_ID?autoplay=0"
  allow="autoplay; fullscreen" allowfullscreen></iframe>
```

---

### 2. Change the button delay

```js
const DELAY = 90; // seconds (90 = 1 min 30 s)
```

---

### 3. Change the CTA destination

```js
btn.addEventListener('click', () => {
  window.location.href = 'YOUR_DESTINATION_URL_HERE';
});
```

---

### 4. Edit copy

| Element | Selector / Location |
|---|---|
| Eyebrow tag | `.eyebrow` text node in `#hero` |
| Headline | `<h1>` in `#hero` |
| Subheadline | `.sub` `<p>` in `#hero` |
| Waiting label | `#ctaLabel` text |
| Button text | Inside `<button id="ctaBtn">` |
| Unlock message | `label.textContent` inside `unlock()` |
